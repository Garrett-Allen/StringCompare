[docs]def jarowinkler(s, t, p=0.1):
    ell = 0
    for i in range(min(4, len(s), len(t))):
        if s[i] == t[i]:
            ell = ell + 1
        else:
            break

    sim = jaro(s, t)

    return sim + ell * p * (1 - sim)
